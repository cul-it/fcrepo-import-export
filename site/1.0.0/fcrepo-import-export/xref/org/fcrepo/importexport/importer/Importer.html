172     /**
173      * Load a {@link BagProfile} for a given directory and initialize the Importer iff the bag passes validation on the
174      * profile and the BagIt spec (handled by {@link #verifyBag}.
175      *
176      * @param bagProfile the name of the {@link BagProfile} to use
177      */
178     private void loadBagProfile(final String bagProfile) {
179         try {
180             final BagProfile profile = config.initBagProfile();
181 
182             final Path root;
183             final File bagDir = config.getBaseDirectory().getAbsoluteFile().getParentFile();
184 
185             // if the bag is serialized (a single file), try to extract first
186             if (bagDir.isFile() && (profile.getSerialization() == BagProfile.Serialization.OPTIONAL ||
187                                     profile.getSerialization() == BagProfile.Serialization.REQUIRED)) {
188                 final BagDeserializer deserializer = SerializationSupport.deserializerFor(bagDir.toPath(), profile);
189                 root = deserializer.deserialize(bagDir.toPath());
190                 // update the base directory so we don't attempt to work on the serialized bag later
191                 config.setBaseDirectory(root.toString());
192             } else {
193                 root = bagDir.toPath();
194             }
195 
196             final Bag bag = verifyBag(root, profile);
197             configureBagItFileMap(bag);
198         } catch (IOException e) {
199             logger.error("Unable to open BagProfile for {}!", bagProfile);
200             throw new RuntimeException(e);
201         }
202     }

226     private void processImport(final URI resource) {
227         importedResources.add(resource);
228 
229         final File importContainerMetadataFile = fileForContainerURI(resource);
230         importContainerDirectory = directoryForContainer(resource);
231 
232         // clean up the membership resources that were imported.
233         membershipResources.clear();
234         discoverMembershipResources(importContainerDirectory);
235 
236         // In order to get the dates right, we need to create resources, *then* set their RDF since
237         // creation of children screws up the dates of parents.
238         try {
239             ensureExists(resource);
240         } catch (FcrepoOperationFailedException | IOException e) {
241             throw new RuntimeException("Unable to create placeholder " + resource, e);
242         }
243 
244         importDirectory(importContainerDirectory);
245 
246         logger.debug("Importing resource {} for file {}", resource, importContainerMetadataFile.getPath());
247 
248         // discover the related resources in the resource being imported.
249         if (!importContainerMetadataFile.exists()) {
250             logger.debug("No container exists in the metadata directory {} for the requested resource {},"
251                     + " importing all contained resources instead.", importContainerMetadataFile.getPath(),
252                     config.getResource());
253         } else {
254             logger.debug("Parsing membership resource in file {}", importContainerMetadataFile.getAbsolutePath());
255             parseMembershipResources(importContainerMetadataFile);
256             importMembershipResources();
257             importRelatedResources();
258             importFile(importContainerMetadataFile);
259         }
260 
261     }

263     private void discoverMembershipResources(final File dir) {
264         if (dir.listFiles() != null) {
265             stream(dir.listFiles()).filter(File::isFile).forEach(this::parseMembershipResources);
266             stream(dir.listFiles()).filter(File::isDirectory).forEach(this::discoverMembershipResources);
267         }
268     }

270     private void parseMembershipResources(final File f) {
271         // skip files that aren't RDF
272         if (!f.getName().endsWith(config.getRdfExtension())) {
273             return;
274         }
275 
276         try {
277             final Model model = parseStream(new FileInputStream(f));
278             if (model.contains(null, MEMBERSHIP_RESOURCE, (RDFNode)null)) {
279                     model.listObjectsOfProperty(MEMBERSHIP_RESOURCE).forEachRemaining(node -> {
280                         logger.info("Membership resource: {}", node);
281                         membershipResources.add(URI.create(node.toString()));
282                     });
283             }
284 
285             // Discover all the related resources with member predicates. Those related resources that aren't imported
286             // during importing the targeted resource are in other container hierarchy that need to handle specifically.
287             // The related resources referenced by default predicate could be ignored.
288             for (final String p : config.getPredicates()) {
289                 if (!p.equals(CONTAINS.toString())) {
290                     for (final NodeIterator it = model.listObjectsOfProperty(createProperty(p)); it.hasNext();) {
291                         final String uri = it.nextNode().toString();
292 
293                         logger.debug("Discovered related resource {} for source {}.", uri, config.getSource());
294                         if (!importedResources.contains(URI.create(uri))) {
295                             // add related resource to list, exclude those that are already imported
296                             final URI resURI = URI.create(uri);
297                             if ((fileForContainerURI(resURI).exists())) {
298                                 relatedResources.add(URI.create(uri));
299 
300                                 logger.debug("Added related resource {}", uri);
301                             } else if ((fileForBinaryURI(resURI).exists() || fileForBinaryURI(resURI)
302                                     .exists())) {
303                                 importedResources.add(URI.create(uri));
304 
305                                 // The binary file will be imported when the non-RDF metadata file is being imported.
306                                 importDirectory(directoryForContainer(resURI));
307                             }
308                         }
309                     }
310                 }
311             }
312         } catch (final IOException e) {
313             throw new RuntimeException("Error reading file: " + f.getAbsolutePath() + ": " + e.toString());
314         } catch (final RiotException e) {
315             throw new RuntimeException("Error parsing RDF: " + f.getAbsolutePath() + ": " + e.toString());
316         }
317     }
318 
319     private void importRelatedResources() {
320         if (relatedResources.size() > 0) {
321             final List<URI> referenceResources = new ArrayList<>(relatedResources);
322             relatedResources.clear();
323             // loop through for nested related resources
324             referenceResources.forEach(uri -> {
325                 logger.info("Importing related resources {} ...", uri);
326                 processImport(uri);
327             });
328         }
329     }

331     private void importMembershipResources() {
332         membershipResources.forEach(this::importMembershipResource);

335     private void importMembershipResource(final URI uri) {
336         final File f = fileForContainerURI(uri);
337         try {
338             final Model diskModel = parseStream(new FileInputStream(f));
339             final Model repoModel = parseStream(client().get(uri).perform().getBody());
340             final FcrepoResponse response = importContainer(uri,
341                                                             sanitize(diskModel.difference(repoModel)),
342                                                             parseHeaders(getHeadersFile(f)));
343             if (response.getStatusCode() == 401) {
344                 importLogger.error("Error importing {} to {}, 401 Unauthorized", f.getAbsolutePath(), uri);
345                 throw new AuthenticationRequiredRuntimeException();
346             } else if (response.getStatusCode() > 204 || response.getStatusCode() < 200) {
347                 importLogger.error("Error importing {} to {}, received {}", f.getAbsolutePath(), uri,
348                     response.getStatusCode());
349                 throw new RuntimeException("Error while importing membership resource " + f.getAbsolutePath()
350                         + " (" + response.getStatusCode() + "): "
351                         + IOUtils.toString(response.getBody(), StandardCharsets.UTF_8));
352             } else {
353                 logger.info("Imported membership resource {}: {}", f.getAbsolutePath(), uri);
354                 importLogger.info("import {} to {}", f.getAbsolutePath(), uri);
355                 successCount.incrementAndGet();
356             }
357         } catch (FcrepoOperationFailedException ex) {
358             importLogger.error(
359                 String.format("Error importing: %1$s to %2$s, Message: %3$s",
360                         f.getAbsolutePath(), uri, ex.getMessage()), ex);
361             throw new RuntimeException("Error importing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
362         } catch (IOException ex) {
363             importLogger.error(
364                 String.format("Error reading/parsing file: %1$s, Message: %2$s",
365                         f.getAbsolutePath(), ex.getMessage()), ex);
366             throw new RuntimeException(
367                     "Error reading or parsing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
368         }
369     }

371     private void importDirectory(final File dir) {
372         // process all the files first (because otherwise they might be
373         // created as peartree nodes which can't be updated with properties
374         // later.
375         if (dir.listFiles() != null) {
376             stream(dir.listFiles()).filter(File::isFile).forEach(this::importFile);
377             stream(dir.listFiles()).filter(File::isDirectory).forEach(this::importDirectory);
378         }

381     private void importFile(final File f) {
382         // The path, relative to the base in the export directory.
383         // This is used in place of the full path to make the output more readable.
384         final String sourceRelativePath =
385                 config.getBaseDirectory().toPath().relativize(f.toPath()).toString();
386         final String filePath = f.getPath();
387 
388         //ignore header files
389         if (filePath.endsWith(".headers")) {
390             // this could be hidden files created by the OS
391             logger.debug("Skipping .headers files ({}).", sourceRelativePath);
392             return;
393 
394         }
395 
396         //parse headers from headers file
397         final Map<String,List<String>> headers = parseHeaders(getHeadersFile(f));
398 
399         //always skip timemaps since they are derived from the mementos they contain.
400         if (isTimeMap(headers)) {
401             logger.debug("Skipping {} :  TimeMaps are never imported.", sourceRelativePath);
402             return;
403         }
404 
405         final boolean isMemento = isMemento(headers);
406 
407         //skip versions when include versions flag is false.
408         if (!config.includeVersions() && isMemento) {
409             logger.debug("Skipping {}: Versions import disabled.", sourceRelativePath);
410             return;
411         }
412 
413         if (filePath.endsWith(BINARY_EXTENSION) || filePath.endsWith(EXTERNAL_RESOURCE_EXTENSION)) {
414             // ... this is only expected to happen when binaries and metadata are written to the same directory...
415 
416             if (!isMemento) {
417                 if (config.isIncludeBinaries()) {
418                     logger.debug("Skipping binary {}: it will be imported when its metadata is imported.",
419                         sourceRelativePath);
420                 } else {
421                     logger.debug("Skipping binary {}", sourceRelativePath);
422                 }
423                 return;
424             } // else continue processing
425 
426         } else if (!filePath.endsWith(config.getRdfExtension())) {
427             // this could be hidden files created by the OS
428             logger.info("Skipping file with unexpected extension ({}).", sourceRelativePath);
429             return;
430         }
431 
432         FcrepoResponse response = null;
433         URI destinationUri = null;
434         try {
435 
436             if (isMemento) {
437                 response = importMemento(f, headers);
438             } else {
439 
440                 final Model model = parseStream(new FileInputStream(f));
441 
442                 // remove the member resources that are being imported
443                 for (final ResIterator it = model.listSubjects(); it.hasNext();) {
444                     final URI uri = URI.create(it.next().toString());
445                     if (relatedResources.contains(uri)) {
446                         relatedResources.remove(uri);
447                     }
448                 }
449 
450                 final ResIterator binaryResources = model.listResourcesWithProperty(RDF_TYPE, NON_RDF_SOURCE);
451                 final ResIterator aclResource = model.listResourcesWithProperty(RDF_TYPE, ACL_SOURCE);
452                 if (binaryResources.hasNext()) {
453                     if (!config.isIncludeBinaries()) {
454                         return;
455                     }
456                     destinationUri = new URI(binaryResources.nextResource().getURI());
457                     logger.info("Importing binary {}", sourceRelativePath);
458                     response = importBinary(destinationUri, model);
459                 } else if (aclResource.hasNext()) {
460                     if (!config.isIncludeAcls()) {
461                         return;
462                     }
463                     destinationUri = new URI(aclResource.nextResource().getURI());
464                     logger.info("Importing acl {}", destinationUri);
465 
466                     final PutBuilder builder = client().put(destinationUri)
467                                                     .body(modelToStream(sanitize(model)), config.getRdfLanguage())
468                                                     .preferLenient();
469                     addInteractionModels(builder, headers);
470                     response = builder.perform();
471                 } else {
472                     destinationUri = uriForFile(f);
473                     if (membershipResources.contains(destinationUri)) {
474                         logger.warn("Skipping Membership Resource: {}", destinationUri);
475                         return;
476                     }
477                     if (model.contains(null, RDF_TYPE, PAIRTREE)) {
478                         logger.info("Skipping PairTree Resource: {}", destinationUri);
479                         return;
480                     }
481 
482                     logger.info("Importing container {} to {}", f.getAbsolutePath(), destinationUri);
483                     response = importContainer(destinationUri, sanitize(model), headers);
484                 }
485             }
486 
487 
488             if (response.getStatusCode() == 401) {
489                 importLogger.error("Error importing {} to {}, 401 Unauthorized", f.getAbsolutePath(),
490                     destinationUri);
491                 throw new AuthenticationRequiredRuntimeException();
492             } else if (response.getStatusCode() > 204 || response.getStatusCode() < 200) {
493                 final String message = "Error while importing " + f.getAbsolutePath() + " ("
494                         + response.getStatusCode() + "): " + IOUtils.toString(response.getBody(),
495                                                                               StandardCharsets.UTF_8);
496                 logger.error(message);
497                 importLogger.error("Error importing {} to {}, received {}", f.getAbsolutePath(), destinationUri,
498                     response.getStatusCode());
499             } else {
500                 logger.info("Imported {}: {}", f.getAbsolutePath(), destinationUri);
501                 importLogger.info("import {} to {}", f.getAbsolutePath(), destinationUri);
502                 successCount.incrementAndGet();
503             }
504         } catch (FcrepoOperationFailedException ex) {
505             importLogger.error(String.format("Error importing %1$s to %2$s, Message: %3$s", f.getAbsolutePath(),
506                 destinationUri, ex.getMessage()), ex);
507             throw new RuntimeException("Error importing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
508         } catch (IOException ex) {
509             importLogger.error(String.format("Error reading/parsing %1$s to %2$s, Message: %3$s",
510                     f.getAbsolutePath(), destinationUri, ex.getMessage()), ex);
511             throw new RuntimeException(
512                     "Error reading or parsing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
513         } catch (URISyntaxException ex) {
514             importLogger.error(
515                 String.format("Error building URI for %1$s, Message: %2$s",
516                         f.getAbsolutePath(), ex.getMessage()), ex);
517             throw new RuntimeException("Error building URI for " + f.getAbsolutePath() + ": " + ex.toString(), ex);
518         }
519     }

521     private FcrepoResponse importMemento(final File mementoFile, final Map<String, List<String>> headers)
522         throws IOException, FcrepoOperationFailedException {
523         final String mementoDatetime = getFirstByKey(headers, MEMENTO_DATETIME_HEADER);
524         final String contentType = getFirstByKey(headers, CONTENT_TYPE_HEADER);
525         final URI timeMapURI = getLinkValueByRel(headers, "timemap");
526         final PostBuilder builder = client().post(timeMapURI)
527             .body(new FileInputStream(mementoFile), contentType)
528             .addHeader(MEMENTO_DATETIME_HEADER, mementoDatetime);
529         return builder.perform();
530     }

532     private boolean hasType(final Map<String, List<String>> headers, final String typeUri) {
533         final List<String> values = headers.get("Link");
534         if (values != null) {
535             for (String linkstr : values) {
536                 final FcrepoLink link = FcrepoLink.valueOf(linkstr);
537                 if (link.getRel().equals("type") && link.getUri().toString().equals(typeUri)) {
538                     return true;
539                 }
540             }
541         }
542         return false;
543     }
544 
545     private boolean isMemento(final Map<String, List<String>> headers) {
546         return hasType(headers, MEMENTO.getURI());
547     }
548 
549     private boolean isTimeMap(final Map<String, List<String>> headers) {
550         return hasType(headers, TIMEMAP.getURI());
551     }
552 
553     private URI getLinkValueByRel(final Map<String, List<String>> headers, final String rel) {
554         final List<String> values = headers.get("Link");
555         for (String linkstr : values) {
556             final FcrepoLink link = FcrepoLink.valueOf(linkstr);
557             if (link.getRel().equals(rel)) {
558                 return link.getUri();
559             }
560         }
561 
562         return null;
563     }

565     private String getFirstByKey(final Map<String, List<String>> headers, final String key) {
566         final List<String> values = headers.get(key);
567         if (values != null && values.size() > 0) {
568             return values.get(0);
569         }
570 
571         return null;
572     }

578     private Map<String, List<String>> parseHeaders(final File headersFile) {
579         try {
580             if (!headersFile.exists()) {
581                 return new HashMap<>();
582             }
583 
584             //converting json to Map
585             final byte[] mapData = Files.readAllBytes(Paths.get(headersFile.toURI()));
586             final ObjectMapper objectMapper = new ObjectMapper();
587             final Map<String, List<String>> headers =

608     private FcrepoResponse importBinary(final URI binaryURI, final Model model)
609             throws FcrepoOperationFailedException, IOException {
610         final String contentType = model.getProperty(createResource(binaryURI.toString()), HAS_MIME_TYPE).getString();
611         final File binaryFile = fileForBinaryURI(binaryURI);
612 
613         final FcrepoResponse binaryResponse = binaryBuilder(binaryURI, binaryFile, contentType, model).perform();
614         if (binaryResponse.getStatusCode() == 201 || binaryResponse.getStatusCode() == 204) {
615             logger.info("Imported binary: {}", binaryURI);
616             importLogger.info("import {} to {}", binaryFile.getAbsolutePath(), binaryURI);
617             successCount.incrementAndGet();

620             return client().put(descriptionURI).body(modelToStream(sanitize(model)), config.getRdfLanguage())
621                 .preferLenient().perform();
622         } else if (binaryResponse.getStatusCode() == 410 && config.overwriteTombstones()) {
623             deleteTombstone(binaryResponse);
624             return binaryBuilder(binaryURI, binaryFile, contentType, model).perform();
625         } else {
626             logger.error("Error while importing {} ({}): {}", binaryFile.getAbsolutePath(),
627                     binaryResponse.getStatusCode(), IOUtils.toString(binaryResponse.getBody(), StandardCharsets.UTF_8));
628             return binaryResponse;
629         }
630     }

632     private PutBuilder binaryBuilder(final URI binaryURI, final File binaryFile, final String contentType,
633             final Model model) throws FcrepoOperationFailedException, IOException {
634         final Map<String, List<String>> headers = parseHeaders(getHeadersFile(binaryFile));
635         String externalContentLocation = null;
636 
637         boolean isRedirect = false;
638         if (headers.containsKey("Location")) {
639             externalContentLocation = getFirstByKey(headers, "Location");
640             isRedirect = true;
641         } else if (headers.containsKey("Content-Location")) {
642             externalContentLocation = getFirstByKey(headers, "Content-Location");
643         }
644 
645         PutBuilder builder = client().put(binaryURI)
646                                      .filename(null);
647 
648         if (externalContentLocation != null) {
649             builder.externalContent(URI.create(externalContentLocation), contentType,
650                                     isRedirect ? "redirect" : "proxy");
651         } else {
652             builder.body(new FileInputStream(binaryFile), contentType).ifUnmodifiedSince(currentTimestamp());
653 
654             if (bagItFileMap != null) {
655                 // Use the bagIt checksum
656                 final String checksum = bagItFileMap.get(binaryFile.getAbsolutePath());
657                 logger.debug("Using Bagit checksum ({}) for file ({}): {}", checksum, binaryFile.getPath(), binaryURI);
658                 builder = builder.digest(checksum, digestAlgorithm);
659             } else {
660                 builder = builder.digestSha1(model.getProperty(createResource(binaryURI.toString()), HAS_MESSAGE_DIGEST)
661                                                   .getObject().toString().replaceAll(".*:",""));
662             }
663         }
664         return builder;
665     }

679     private PutBuilder containerBuilder(final URI uri, final Model model, final Map<String,List<String>> headers) {
680         PutBuilder builder = client().put(uri)
681                                      .body(modelToStream(model), config.getRdfLanguage())
682                                      .ifUnmodifiedSince(currentTimestamp());
683         if (bagItFileMap != null && config.getBagProfile() != null) {
684             // Use the bagIt checksum
685             final File containerFile = Paths.get(fileForContainerURI(uri).toURI()).normalize().toFile();
686             final String checksum = bagItFileMap.get(containerFile.getAbsolutePath());
687             logger.debug("Using Bagit checksum ({}) for file ({})", checksum, containerFile.getPath());
688             builder = builder.digest(checksum, digestAlgorithm);
689         }
690 
691         addInteractionModels(builder, headers);
692         return builder;

695     private void addInteractionModels(final PutBuilder builder, final Map<String, List<String>> headers) {
696         headers.entrySet().stream().filter(entry -> entry.getKey().equals("Link"))
697             .flatMap(entry -> entry.getValue().stream())
698             .forEach(linkstr -> {

701                     final String interactionModel = link.getUri().toString();
702                     if (INTERACTION_MODELS.contains(interactionModel)) {
703                         builder.addInteractionModel(interactionModel);
704                     }
705                 }
706             });

727     /**
728      * Removes statements from the provided model that affect triples that need not be (and indeed
729      * cannot be) modified directly through PUT, POST or PATCH requests to fedora.
730      *
731      * Certain triples included in a resource from fedora cannot be explicitly stored, but because
732      * they're derived from other content that *can* be stored will still appear identical when the
733      * other RDF and content is ingested.  Examples include those properties that reflect innate
734      * characteristics of binary resources like file size and message digest,  Or triples that
735      * represent characteristics of rdf resources like the number of children, whether it has
736      * versions and some of the types.
737      *
738      * @param model the RDF statements about an exported resource
739      * @return the provided model updated to omit statements that may not be updated directly through
740      *         the fedora API
741      * @throws IOException
742      * @throws FcrepoOperationFailedException
743      */
744     private Model sanitize(final Model model) throws IOException, FcrepoOperationFailedException {
745         final List<Statement> remove = new ArrayList<>();
746         for (final StmtIterator it = model.listStatements(); it.hasNext(); ) {
747             final Statement s = it.nextStatement();
748 
749             if ((s.getPredicate().getNameSpace().equals(REPOSITORY_NAMESPACE) && !relaxedPredicate(s.getPredicate()))
750                     || s.getSubject().getURI().endsWith("fcr:export?format=jcr/xml")
751                     || s.getSubject().getURI().equals(REPOSITORY_NAMESPACE + "jcr/xml")
752                     || s.getPredicate().equals(DESCRIBEDBY)
753                     || s.getPredicate().equals(CONTAINS)
754                     || s.getPredicate().equals(HAS_MESSAGE_DIGEST)
755                     || (s.getPredicate().equals(RDF_TYPE) && forbiddenType(s.getResource()))) {
756                 remove.add(s);
757             } else if (s.getObject().isResource()) {
758                 // make sure that referenced repository objects exist
759                 final String obj = s.getResource().toString();
760                 if (obj.startsWith(repositoryRoot.toString())) {
761                     ensureExists(URI.create(obj));
762                 }
763             }
764         }
765         return model.remove(remove);
766     }

905     /**
906      * Query a {@link Bag} for the highest ranking {@link Manifest} and use that in order to populate the
907      * {@code bagItFileMap} and {@code digestAlgorithm} for use when importing files
908      *
909      * @param bag The {@link Bag} to read from
910      */
911     private void configureBagItFileMap(final Bag bag) {
912         // The fcrepo-client-java only supports up to sha256 so we only check against each of md5, sha1, and sha256
913         final BagItDigest md5 = BagItDigest.MD5;
914         final BagItDigest sha = BagItDigest.SHA1;
915         final BagItDigest sha256 = BagItDigest.SHA256;
916         final Set<String> fcrepoSupported = new HashSet<>(Arrays.asList(md5.bagitName(), sha.bagitName(),
917                                                                         sha256.bagitName()));
918         final Manifest manifest = bag.getPayLoadManifests().stream()
919                .filter(streamManifest -> fcrepoSupported.contains(streamManifest.getAlgorithm().getBagitName()))
920                .reduce((m1, m2) -> manifestPriority(m1) > manifestPriority(m2) ? m1 : m2)
921                .orElseThrow(() -> new RuntimeException("Bag does not contain any manifests the import " +
922                                                        "utility can use! Available algorithms are: " +
923                                                        StringUtils.join(fcrepoSupported, ",")));
924 
925         this.bagItFileMap = manifest.getFileToChecksumMap().entrySet().stream()
926                                     .collect(Collectors.toMap(
927                                         entry -> entry.getKey().toAbsolutePath().toString(),
928                                         Map.Entry::getValue));
929         logger.debug("loaded checksum map: {}", bagItFileMap);
930 
931         switch(BagItDigest.from(manifest.getAlgorithm().getBagitName())) {
932             case MD5:
933                 this.digestAlgorithm = md5.bagitName();
934                 break;
935             case SHA1:
936                 // fcrepo-client expects only "sha" for sha1 headers
937                 this.digestAlgorithm = "sha";
938                 break;
939             case SHA256:
940                 this.digestAlgorithm = sha256.bagitName();
941                 break;
942             default: throw new RuntimeException("Unsupported BagIt algorithm!");
943         }
944     }

946     /**
947      * Receive the priority for a given {@link Manifest} based on its digest algorithm
948      *
949      * @param manifest the manifest to prioritize
950      * @return the priority of the digest algorithm
951      */
952     private int manifestPriority(final Manifest manifest) {
953         final String bagItAlgorithm = manifest.getAlgorithm().getBagitName();
954         switch (BagItDigest.from(bagItAlgorithm)) {
955             case MD5: return 0;
956             case SHA1: return 1;
957             case SHA256: return 2;
958             default: throw new RuntimeException("Algorithm not allowed! " + bagItAlgorithm);
959         }

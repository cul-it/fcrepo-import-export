282    /**
283     * Sets the URI map, for mapping URIs being imported
284     *
285     * @param map Array containing two URIs for the export baseURL and the import baseURL
286     */
287    public void setMap(final String[] map) {
288        if (map.length == 2 && map[0] != null && map[1] != null) {
289            final String sourceStr = map[0];
290            final String destinationStr = map[1];
291            if ((sourceStr.endsWith("/") && !destinationStr.endsWith("/")) ||
292                    (!sourceStr.endsWith("/") && destinationStr.endsWith("/"))) {
293                logger.warn("Possible mismatch between the source and destination URIs: one ended with a trailing "
294                        + "slash but the other did not: \"{}\" -> \"{}\" (dropping the trailing slash)", sourceStr,
295                        destinationStr);
296                this.source = URI.create(removeTrailingSlash(sourceStr));
297                this.destination = URI.create(removeTrailingSlash(destinationStr));
298            } else {
299                this.source = URI.create(sourceStr);
300                this.destination = URI.create(destinationStr);
301            }
302
303        } else {
304            throw new IllegalArgumentException("The map should contain the export and import baseURLs");
305        }
306    }
307
308    private String removeTrailingSlash(final String url) {
309        if (url.endsWith("/")) {
310            return url.substring(0, url.lastIndexOf('/'));
311        } else {
312            return url;
313        }
314    }

376    /**
377     * Try to initialize a BagProfile for either the Importer or Exporter. If the bag profile passed in is "default",
378     * use the fedora-import-export identifier.
379     *
380     * @return the {@link BagProfile}
381     * @throws IOException if the {@link BagProfile} cannot be initialized
382     */
383    public BagProfile initBagProfile() throws IOException {
384        final BagProfile.BuiltIn builtIn = "default".equalsIgnoreCase(bagProfile)
385                                           ? BagProfile.BuiltIn.FEDORA_IMPORT_EXPORT
386                                           : BagProfile.BuiltIn.from(bagProfile);
387        return new BagProfile(builtIn);
388    }

560    /**
561     * Generate a HashMap suitable for serializing to Yaml
562     *
563     * @return Map key value pairs of configuration
564     */
565    public Map<String, String> getMap() {
566        final Map<String, String> map = new HashMap<String, String>();
567        map.put("mode", (this.isImport() ? "import" : "export"));
568        map.put("resource", this.getResource().toString());
569        if (this.getSource() != null && this.getDestination() != null) {
570            map.put("map", this.getSource() + "," + this.getDestination());
571        }
572        map.put("dir", this.baseDirectory.getAbsolutePath());
573        if (!this.getRdfLanguage().isEmpty()) {
574            map.put("rdfLang", this.getRdfLanguage());
575        }
576        map.put("acls", Boolean.toString(this.includeAcls));
577        map.put("binaries", Boolean.toString(this.includeBinaries));
578        map.put("external", Boolean.toString(this.retrieveExternal));
579        map.put("inbound", Boolean.toString(this.retrieveInbound));
580        map.put("membership", Boolean.toString(this.includeMembership));
581        map.put("overwriteTombstones", Boolean.toString(this.overwriteTombstones()));
582        map.put("legacyMode", Boolean.toString(this.isLegacy()));
583        map.put("versions", Boolean.toString(this.includeVersions));
584        if (this.getBagProfile() != null) {
585            map.put("bag-profile", this.getBagProfile());
586        }
587        if (this.getBagConfigPath() != null) {
588            map.put("bag-config", this.getBagConfigPath());
589        }
590        if (this.getBagSerialization() != null) {
591            map.put("bag-serialization", this.getBagSerialization());
592        }
593        if (this.getBagAlgorithms() != null) {
594            map.put("bag-algorithms", String.join(",", this.getBagAlgorithms()));
595        }
596        final String predicates = String.join(",", this.getPredicates());
597        map.put("predicates", predicates);
598        map.put("auditLog", Boolean.toString(this.auditLog));
599        return map;
600    }

602    /**
603     * Get extension for the RDF language provided
604     *
605     * @param language RDF language mimetype
606     * @return string of file extension
607     */
608    private static String getRDFExtension(final String language) {
609        final Lang lang = contentTypeToLang(language);
610        if (lang == null) {
611            throw new RuntimeException(language + " is not a recognized RDF language");
612        }
613
614        return "." + lang.getFileExtensions().get(0);
615
616    }

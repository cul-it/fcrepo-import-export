152    private void configureBagItParameters() {
153        try {
154            // parse config + load profile
155            final BagConfig bagConfig = loadBagConfig(config.getBagConfigPath());
156            final BagProfile bagProfile = config.initBagProfile();
157
158            // configure the BagIt algorithms to use + setup the fields for the Exporter
159            final Set<BagItDigest> algorithms = setupBagItDigests(bagProfile);
160
161            // enforce default metadata
162            bagProfile.validateConfig(bagConfig);
163            final Map<String, String> profileMetadata = bagProfile.getProfileMetadata();
164
165            // the profile identifier must exist per the bagit-profiles spec
166            bagProfileId = profileMetadata.get(BagProfileConstants.BAGIT_PROFILE_IDENTIFIER);
167
168            // check if serialization is required
169            final String serializationFormat = config.getBagSerialization();
170            if (serializationFormat != null) {
171                // this can throw exceptions if the serialization format is not supported
172                bagSerializer = SerializationSupport.serializerFor(serializationFormat, bagProfile);
173            }
174
175            // setup bag
176            final File bagdir = config.getBaseDirectory().getParentFile();
177            this.bag = new BagWriter(bagdir, algorithms);
178            for (final String tagFile : bagConfig.getTagFiles()) {
179                this.bag.addTags(tagFile, bagConfig.getFieldsForTagFile(tagFile));
180            }
181        } catch (FileNotFoundException e) {
182            throw new RuntimeException(String.format("Error loading bag config file: %1$s", e.getMessage()), e);
183        } catch (IOException e) {
184            throw new RuntimeException(String.format("Error reading bag profile: %1$s", e.getMessage()), e);
185        }
186    }

188    /**
189     * Set up the hash algorithms (as {@link BagItDigest}) to use based on the {@link Config} and {@link BagProfile}
190     * Always use the user specified + profile required algorithms
191     * If neither is specified, use the profile "manifest-allowed" and "tag-manifest-allowed"
192     * If the algorithms are still empty, use sha1
193     *
194     * @param profile the profile to get BagItDigests for
195     * @return the algorithms which were initialized, as {@link BagItDigest}s
196     */
197    private Set<BagItDigest> setupBagItDigests(final BagProfile profile) {
198        final Set<String> algorithms = new HashSet<>();
199        final Set<String> allowedAlgorithms = profile.getAllowedPayloadAlgorithms();
200        final Set<String> requiredAlgorithms = profile.getPayloadDigestAlgorithms();
201
202        // first validate that the profile supports user specified algorithms
203        final String[] userAlgorithms = config.getBagAlgorithms();
204        if (userAlgorithms != null) {
205            for (String algorithm : userAlgorithms) {
206                if (!allowedAlgorithms.isEmpty() && !allowedAlgorithms.contains(algorithm)) {
207                    throw new RuntimeException("Bag Profile does not allow specified algorithm: " + algorithm +
208                                               ". Allowed algorithms are: " + allowedAlgorithms);
209                }
210
211                algorithms.add(algorithm);
212            }
213        }
214
215        // always add required algorithms
216        algorithms.addAll(requiredAlgorithms);
217
218        // check if we should fallback to the allowed algorithms or sha1
219        if (algorithms.isEmpty() && !allowedAlgorithms.isEmpty()) {
220            algorithms.addAll(allowedAlgorithms);
221        } else if (algorithms.isEmpty()) {
222            algorithms.add(BagItDigest.SHA1.bagitName());
223        }
224
225        return algorithms.stream()
226                             .map(String::toUpperCase)
227                             .map(BagItDigest::valueOf)
228                             .peek(this::setupFileMap)
229                             .collect(Collectors.toSet());
230    }

232    /**
233     * Configure the digest algorithm and fileMap for a given BagItDigest
234     *
235     * @param digest the BagItDigest to initialize fields for
236     */
237    private void setupFileMap(final BagItDigest digest) {
238        switch (digest) {
239            case MD5:
240                this.md5 = digest.messageDigest();
241                this.md5FileMap = new HashMap<>();
242                break;
243            case SHA1:
244                this.sha1 = digest.messageDigest();
245                this.sha1FileMap = new HashMap<>();
246                break;
247            case SHA256:
248                this.sha256 = digest.messageDigest();
249                this.sha256FileMap = new HashMap<>();
250                break;
251            case SHA512:
252                this.sha512 = digest.messageDigest();
253                this.sha512FileMap = new HashMap<>();
254                break;
255            default:
256                throw new IllegalStateException("Unexpected BagIt algorithm: " + digest);
257        }
258    }

283    /**
284     * This method does the export
285     */
286    @Override
287    public void run() {
288        logger.info("Running exporter...");
289        if (repositoryRoot == null) {
290            try {
291                logger.info("Attempting to automatically determine the repository root");
292                findRepositoryRoot(config.getResource());
293            } catch (IOException | FcrepoOperationFailedException e) {
294                throw new RuntimeException("Failed to locate the root of the repository being exported", e);
295            }
296        }
297        logger.debug("Repository root is " + repositoryRoot);
298
299        export(config.getResource());
300        if (bag != null) {
301            try {
302                logger.info("Finishing bag manifests...");
303                bag.addTags(BagConfig.BAG_INFO_KEY, bagTechMetadata());
304                if (sha1 != null) {
305                    bag.registerChecksums(BagItDigest.SHA1, sha1FileMap);
306                }
307                if (sha256 != null) {
308                    bag.registerChecksums(BagItDigest.SHA256, sha256FileMap);
309                }
310                if (sha512 != null) {
311                    bag.registerChecksums(BagItDigest.SHA512, sha512FileMap);
312                }
313                if (md5 != null) {
314                    bag.registerChecksums(BagItDigest.MD5, md5FileMap);
315                }
316                bag.write();
317
318                if (bagSerializer != null) {
319                    // Make sure the path is an absolute path because the BagSerializer uses Path#relativize which
320                    // requires both Paths to be of the same type
321                    bagSerializer.serialize(bag.getRootDir().getAbsoluteFile().toPath());
322                }
323            } catch (IOException e) {
324                throw new RuntimeException("Error finishing Bag: " + e.toString());
325            } catch (Exception e) {
326                e.printStackTrace();
327            }
328        }
329        exportLogger.info("Finished export... {} bytes/{} resources exported", successBytes.get(), successCount.get());
330    }

332    private Map<String, String> bagTechMetadata() {
333        final Map<String, String> metadata = new HashMap<>();
334        metadata.put(BagProfileConstants.BAGIT_PROFILE_IDENTIFIER, bagProfileId);
335        metadata.put(BagConfig.BAG_SIZE_KEY, byteCountToDisplaySize(successBytes.longValue()));
336        metadata.put(BagConfig.PAYLOAD_OXUM_KEY, successBytes.toString() + "." + successCount.toString());
337        metadata.put(BagConfig.BAGGING_DATE_KEY, dateFormat.format(new Date()));
338        return metadata;

341    private void export(final URI uri) {
342        logger.trace("HEAD " + uri);
343        try (FcrepoResponse response = client().head(uri).disableRedirects().perform()) {
344            if (response.getStatusCode() == 404 && uri.toString().endsWith("fcr:acl")) {
345                logger.debug("ACL {} not found and thus will not be exported.", uri);
346                return;
347            }
348
349            checkValidResponse(response, uri, config.getUsername());
350            final List<URI> linkHeaders = response.getLinkHeaders("type");
351            final URI acl = response.getLinkHeaders("acl").stream().findFirst().orElse(null);
352
353            if (linkHeaders.contains(binaryURI)) {
354                logger.debug("Found binary at " + uri);
355                final boolean external = response.getHeaderValue("Content-Location") != null;
356                final List<URI> describedby = response.getLinkHeaders("describedby");
357                exportBinary(uri, describedby, external);
358            } else if (linkHeaders.contains(containerURI) || linkHeaders.contains(rdfSourceURI)) {
359                logger.debug("Found container at " + uri);
360                exportRdf(uri, null);
361                // Export versions for this container
362                exportVersions(uri);
363            } else if (uri.equals(URI.create(repositoryRoot.toString() + "/fcr:acl").normalize())) {
364                logger.info("The repository default root ACL is not being exported: {}", uri);
365            } else {
366                logger.error("Resource is not an LDP Container, LDP RDFSource,  or an LDP NonRDFSource: {}", uri);
367                exportLogger.error("Resource is not an LDP Container, LDP RDFSource, or an LDP NonRDFSource: {}", uri);
368            }
369
370            if (acl != null && config.isIncludeAcls()) {
371                export(acl);
372            }
373
374
375        } catch (FcrepoOperationFailedException ex) {
376            logger.warn("Error retrieving content: {}", ex.toString());
377            exportLogger.error(String.format("Error retrieving context of uri: %1$s, Message: %2$s",
378                    uri, ex.toString()),
379                ex);
380        } catch (IOException ex) {
381            logger.warn("Error writing content: {}", ex.toString());
382            exportLogger.error(String.format("Error writing content from uri: %1$s, Message: %2$s", uri, ex.toString()),
383                ex);
384        }
385    }

387    private void exportBinary(final URI uri, final List<URI> describedby, final boolean external)
388            throws FcrepoOperationFailedException, IOException {
389
390        if (!config.isIncludeBinaries()) {
391            logger.info("Skipping {}", uri);
392            return;
393        }
394
395        GetBuilder getBuilder = client().get(uri);
396        if (external && !config.retrieveExternal()) {
397            getBuilder = getBuilder.disableRedirects();
398        }
399        try (FcrepoResponse response = getBuilder.perform()) {
400            checkValidResponse(response, uri, config.getUsername());
401
402            final File file = external ? fileForExternalResources(uri, null, null, config.getBaseDirectory()) :
403                    fileForBinary(uri, null, null, config.getBaseDirectory());
404
405            logger.info("Exporting binary: {}", uri);
406            writeResponse(uri, response.getBody(), describedby, file);
407            writeHeadersFile(response, getHeadersFile(file));
408            exportLogger.info("export {} to {}", uri, file.getAbsolutePath());
409            successCount.incrementAndGet();
410
411        }
412
413        // Export versions for this binary
414        exportVersions(uri);
415    }

417    private void exportRdf(final URI uri, final URI binaryURI)
418            throws FcrepoOperationFailedException {
419        final File file = fileForURI(uri, null, null, config.getBaseDirectory(), config.getRdfExtension());
420        if (file == null) {
421            logger.info("Skipping {}", uri);
422            return;
423        } else if (file.exists()) {
424            logger.info("Already exported {}", uri);
425            return;
426        }
427
428        final GetBuilder getBuilder = client().get(uri).accept(config.getRdfLanguage());
429
430        final List<URI> includeUris = new ArrayList<>();
431        final List<URI> omitUris = new ArrayList<>();
432
433        if (config.retrieveInbound()) {
434            includeUris.add(URI.create(INBOUND_REFERENCES.getURI()));
435        }
436        if (!config.includeMembership()) {
437            omitUris.add(URI.create(PREFER_MEMBERSHIP.getURI()));
438        }
439
440        getBuilder.preferRepresentation(includeUris, omitUris);
441
442        try (FcrepoResponse response = getBuilder.perform()) {
443            checkValidResponse(response, uri, config.getUsername());
444            logger.info("Exporting rdf: {}", uri);
445
446            final String responseBody = IOUtils.toString(response.getBody(), StandardCharsets.UTF_8);
447            final Model model = createDefaultModel().read(new ByteArrayInputStream(responseBody.getBytes()),
448                    null, config.getRdfLanguage());
449            Set<URI> inboundMembers = null;
450
451            if (!config.isIncludeBinaries() || config.retrieveInbound()) {
452
453                if (!config.isIncludeBinaries()) {
454                    filterBinaryReferences(uri, model);
455                }
456
457                if (config.retrieveInbound()) {
458                    final URI subject = (binaryURI != null) ? binaryURI : uri;
459                    inboundMembers = filterInboundReferences(subject, model);
460                }
461
462                try (final ByteArrayOutputStream out = new ByteArrayOutputStream()) {
463                    RDFDataMgr.write(out, model, contentTypeToLang(config.getRdfLanguage()));
464                    writeResponse(uri, new ByteArrayInputStream(out.toByteArray()), null, file);
465                }
466            } else {
467                // we can write the body to disk unfiltered
468                writeResponse(uri, new ByteArrayInputStream(responseBody.getBytes()), null, file);
469            }
470
471            //write headers file
472            writeHeadersFile(response, getHeadersFile(file));
473
474            exportLogger.info("export {} to {}", uri, file.getAbsolutePath());
475            successCount.incrementAndGet();
476
477            exportMembers(model, inboundMembers);
478            exportVersions(uri);
479        } catch ( Exception ex ) {
480            ex.printStackTrace();
481            exportLogger.error(String.format("Error exporting description: %1$s, Cause: %2$s",
482                    uri, ex.getMessage()), ex);
483        }
484
485    }

487    private File getHeadersFile(final File file) {
488        return new File(file.getParentFile(), file.getName() + HEADERS_EXTENSION);
489    }
490
491    void writeHeadersFile(final FcrepoResponse response, final File file) throws IOException {
492
493        final Map<String, List<String>> headers = response.getHeaders();
494        if (!headers.isEmpty()) {
495            final String json = new ObjectMapper().writeValueAsString(headers);
496            final InputStream byteInputStream = new ByteArrayInputStream(json.getBytes());
497            try (OutputStream headerOut = Files.newOutputStream(file.toPath())) {
498                copy(byteInputStream, headerOut);
499                if (md5FileMap != null) {
500                    md5FileMap.put(file, new String(encodeHex(md5.digest())));
501                }
502                if (sha1FileMap != null) {
503                    sha1FileMap.put(file, new String(encodeHex(sha1.digest())));
504                }
505                if (sha256FileMap != null) {
506                    sha256FileMap.put(file, new String(encodeHex(sha256.digest())));
507                }
508                if (sha512FileMap != null) {
509                    sha512FileMap.put(file, new String(encodeHex(sha512.digest())));
510                }
511            }
512        }
513    }

515    private Set<URI> filterInboundReferences(final URI uri, final Model model) {
516        final String withSlash = withSlash(uri).toString();
517        final String withoutSlash = withoutSlash(uri).toString();
518        final Set<URI> inboundMembers = new HashSet<>();
519        final List<Statement> removeList = new ArrayList<>();
520        for (final StmtIterator inbound = model.listStatements(); inbound.hasNext(); ) {
521            final Statement s = inbound.next();
522            final String subject = s.getSubject().toString();
523            if (!subject.equals(withSlash) && !subject.equals(withoutSlash)) {
524                removeList.add(s);
525                logger.trace("Filtering inbound reference: {}", s);
526                inboundMembers.add(URI.create(subject));
527            }
528        }
529
530        model.remove(removeList);
531        return inboundMembers;
532    }
533
534    private void exportMembers(final Model model, final Set<URI> inboundMembers) {
535        for (final String p : config.getPredicates()) {
536            final NodeIterator members = model.listObjectsOfProperty(createProperty(p));
537            while (members.hasNext()) {
538                export(URI.create(members.nextNode().toString()));
539            }
540        }
541
542        if (inboundMembers != null) {
543            for (final URI inbound : inboundMembers) {
544                export(inbound);
545            }
546        }
547    }

549    /**
550     * Filter out the binary resource references from the model
551     * @param uri the URI for the resource
552     * @param model the RDF Model of the resource
553     * @return the RDF model with no binary references
554     * @throws FcrepoOperationFailedException
555     * @throws IOException
556     */
557    private void filterBinaryReferences(final URI uri, final Model model) throws IOException,
558            FcrepoOperationFailedException {
559
560        final List<Statement> removeList = new ArrayList<>();
561        for (final StmtIterator it = model.listStatements(); it.hasNext();) {
562            final Statement s = it.nextStatement();
563
564            final RDFNode obj = s.getObject();
565            if (obj.isResource() && obj.toString().startsWith(repositoryRoot.toString())
566                    && !s.getPredicate().toString().equals(REPOSITORY_NAMESPACE + "hasTransactionProvider")) {
567                try (final FcrepoResponse resp = client().head(URI.create(obj.toString())).disableRedirects()
568                        .perform()) {
569                    checkValidResponse(resp, URI.create(obj.toString()), config.getUsername());
570                    final List<URI> linkHeaders = resp.getLinkHeaders("type");
571                    if (linkHeaders.contains(binaryURI)) {
572                        removeList.add(s);
573                    }
574                }
575            }
576        }
577
578        model.remove(removeList);
579    }

584     * @throws IOException If an I/O error occurs

587    private void findRepositoryRoot(final URI uri) throws IOException, FcrepoOperationFailedException {

590        if (!isRepositoryRoot(uri, client(), config)) {
591            findRepositoryRoot(URI.create(repositoryRoot.toString().substring(0,
592                    repositoryRoot.toString().lastIndexOf("/"))));
593        }

596    /**
597     * Initiates export of versions for the given resource if it is a versioned resourced
598     * 
599     * @param uri resource uri
600     * @throws FcrepoOperationFailedException
601     * @throws IOException
602     */
603    private void exportVersions(final URI uri) throws FcrepoOperationFailedException, IOException {
604        // Do not check for versions if disabled, already exporting a version, or the repo root
605        if (!config.includeVersions() || uri.equals(repositoryRoot)) {
606            return;
607        }
608
609        // Resolve the timemap endpoint for this resource
610        final URI timemapURI;
611        try (FcrepoResponse response = client().head(uri).disableRedirects().perform()) {
612            checkValidResponse(response, uri, config.getUsername());
613            if (response.getLinkHeaders("type").contains(URI.create(MEMENTO.toString()))) {
614                logger.trace("Resource {} is a memento and therefore not versioned:  ", uri);
615                return;
616            } else if (response.getLinkHeaders("type").contains(URI.create(TIMEMAP.toString()))) {
617                logger.trace("Resource {} is a timemap and therefore not versioned:  ", uri);
618                return;
619            }
620
621            timemapURI = response.getLinkHeaders("timemap").stream().findFirst().orElse(null);
622            if (timemapURI == null) {
623                logger.trace("Resource {} is not versioned:  no rel=\"timemap\" Link header present on {}", uri);
624                return;
625            }
626        }
627
628        export(timemapURI);
629
630        try (FcrepoResponse response = client().get(timemapURI).accept(config.getRdfLanguage()).perform()) {
631           // Verify that timemapURI can be accessed, which will fail if the resource is not versioned
632            checkValidResponse(response, timemapURI, config.getUsername());
633            // Extract uris of mementos for export
634            final Model model = createDefaultModel().read(response.getBody(), null, config.getRdfLanguage());
635            final StmtIterator versionsIt = model.listStatements();
636            while (versionsIt.hasNext()) {
637                final Statement versionSt = versionsIt.next();
638                if (versionSt.getPredicate().equals(CONTAINS)) {
639                    final Resource versionResc = versionSt.getResource();
640                    exportLogger.info("Exporting version: {}", versionResc.getURI());
641                    logger.info("Exporting version {} for {}", versionResc.getURI(), uri);
642                    export(URI.create(versionResc.getURI()));
643                }
644            }
645        }
646    }

648    private URI addRelativePath(final URI uri, final String path) {
649        final String base = uri.toString();
650
651        if (base.charAt(base.length() - 1) == '/') {
652            if (path.charAt(0) == '/') {
653                return URI.create(base + path.substring(1, path.length()));
654            }
655            return URI.create(base + path);
656        } else if (path.charAt(0) == '/') {
657            return URI.create(base + path);
658        }
659
660        return URI.create(base + "/" + path);
661    }

663    void writeResponse(final URI uri, final InputStream in, final List<URI> describedby, final File file)
664            throws IOException, FcrepoOperationFailedException {
665        if (!file.getParentFile().exists()) {
666            file.getParentFile().mkdirs();
667        }
668        try (OutputStream out = new FileOutputStream(file)) {
669            copy(in, out);
670            logger.info("Exported {} to {}", uri, file.getAbsolutePath());
671
672            if (md5FileMap != null) {
673                md5FileMap.put(file, new String(encodeHex(md5.digest())));
674            }
675            if (sha1FileMap != null) {
676                sha1FileMap.put(file, new String(encodeHex(sha1.digest())));
677            }
678            if (sha256FileMap != null) {
679                sha256FileMap.put(file, new String(encodeHex(sha256.digest())));
680            }
681            if (sha512FileMap != null) {
682                sha512FileMap.put(file, new String(encodeHex(sha512.digest())));
683            }
684        }
685
686        if (describedby != null) {
687            for (final Iterator<URI> it = describedby.iterator(); it.hasNext(); ) {
688                exportRdf(it.next(), uri);
689            }
690        }
691    }

693    /**
694     * Copy bytes and generate checksums
695     * @param in Source data
696     * @param out Data destination
697     * @throws IOException If an I/O error occurs
698     */
699    private void copy(final InputStream in, final OutputStream out) throws IOException {
700        if (md5 != null) {
701            md5.reset();
702        }
703        if (sha1 != null) {
704            sha1.reset();
705        }
706        if (sha256 != null) {
707            sha256.reset();
708        }
709        if (sha512 != null) {
710            sha512.reset();
711        }
712
713        int read = 0;
714        final byte[] buf = new byte[8192];
715        while ((read = in.read(buf)) != -1) {
716            if (md5 != null) {
717                md5.update(buf, 0, read);
718            }
719            if (sha1 != null) {
720                sha1.update(buf, 0, read);
721            }
722            if (sha256 != null) {
723                sha256.update(buf, 0, read);
724            }
725            if (sha512 != null) {
726                sha512.update(buf, 0, read);
727            }
728            if (out != null) {
729                out.write(buf, 0, read);
730                successBytes.addAndGet(read);
731            }
732        }
733    }

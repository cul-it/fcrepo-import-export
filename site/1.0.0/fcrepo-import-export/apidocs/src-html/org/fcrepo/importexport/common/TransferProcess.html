021import static org.fcrepo.importexport.common.FcrepoConstants.BINARY_EXTENSION;
022import static org.fcrepo.importexport.common.FcrepoConstants.EXTERNAL_RESOURCE_EXTENSION;
023import static org.fcrepo.importexport.common.FcrepoConstants.NON_RDF_SOURCE;
024import static org.fcrepo.importexport.common.FcrepoConstants.RDF_TYPE;
025import static org.fcrepo.importexport.common.FcrepoConstants.REPOSITORY_ROOT;

027import java.io.File;
028import java.io.IOException;
029import java.io.UnsupportedEncodingException;
030import java.net.URI;
031import java.net.URLDecoder;
032import java.net.URLEncoder;
033
034import org.apache.jena.rdf.model.Model;
035import org.fcrepo.client.FcrepoClient;

040/**
041 * @author lsitu
042 * @author barmintor
043 * @since 2016-08-31
044 */
045public interface TransferProcess {
046
047    final static String IMPORT_EXPORT_LOG_PREFIX = "org.fcrepo.importexport.audit";
048    final static String BAGIT_CHECKSUM_DELIMITER = "  ";
049
050    // Used only to load patched RDF writers
051    RdfWriterHelper notUsed = new RdfWriterHelper();
052
053    /**
054     * This method does the import or export
055     */
056    public void run();
057
058    /**
059     * Encodes a path (as if from a URI) to avoid
060     * characters that may be disallowed in a filename.
061     * This operation can be reversed by invoking
062     * {@link #decodePath }.
063     * @param path the path portion of a URI
064     * @return a version of the path that avoid characters
065     * such as ":".
066     */
067    public static String encodePath(final String path) {
068        try {
069            return URLEncoder.encode(path, "UTF-8").replace("%2F", "/");
070        } catch (UnsupportedEncodingException e) {
071            throw new RuntimeException(e);
072        }
073    }
074
075    /**
076     * Decodes a path. This operation can be reversed by
077     * invoking {@link #encodePath }.
078     * @param encoded the path portion of a URI
079     * @return the original path
080     */
081    public static String decodePath(final String encoded) {
082        try {
083            return URLDecoder.decode(encoded, "UTF-8");
084        } catch (UnsupportedEncodingException e) {
085            throw new RuntimeException(e);
086        }
087    }
088
089    /**
090     * Gets the file where a binary resource at the given URL would be stored
091     * in the export package.
092     * @param uri the URI for the resource
093     * @param sourcePath the base path the resource was exported from
094     * @param destinationPath the base path the resource is importing to
095     * @param baseDir the base directory in the export package
096     * @return a unique location for binary resources from the path of the given URI
097     *         would be stored
098     */
099    public static File fileForBinary(final URI uri, final String sourcePath, final String destinationPath,
100            final File baseDir) {
101        return fileForURI(uri, sourcePath, destinationPath, baseDir, BINARY_EXTENSION);
102    }
103
104    /**
105     * Gets the file where an external binary resource at the given URL would be stored
106     * in the export package.
107     * @param uri the URI for the resource
108     * @param sourcePath the base path the resource was exported from
109     * @param destinationPath the base path the resource is importing to
110     * @param baseDir the base directory in the export package
111     * @return a unique location for external resources from the path of the given URI
112     *         would be stored
113     */
114    public static File fileForExternalResources(final URI uri, final String sourcePath,
115            final String destinationPath, final File baseDir) {
116        return fileForURI(uri, sourcePath, destinationPath, baseDir, EXTERNAL_RESOURCE_EXTENSION);
117    }
118
119    /**
120     * Gets the file where a resource at the given URL with the given extension
121     * would be stored in the export package.
122     * @param uri the URI for the resource
123     * @param sourcePath the base path the resource was exported from
124     * @param destinationPath the base path the resource is importing to
125     * @param baseDir the baseDir directory in the export package
126     * @param extension the arbitrary extension expected the file
127     * @return a unique location for resources from the path of the given URI
128     *         would be stored
129     */
130    public static File fileForURI(final URI uri, final String sourcePath, final String destinationPath,
131            final File baseDir, final String extension) {
132        String path = uri.getPath();
133        if (sourcePath != null && destinationPath != null) {
134            path = path.replaceFirst(destinationPath, sourcePath);
135        }
136        if (extension != null && path.endsWith("/")) {
137            path = path.substring(0, path.length() - 1);
138        }
139        return new File(baseDir, encodePath(path) + extension);
140    }
141
142    /**
143     * Gets the directory where metadata resources contained by the resource at the given
144     * URI would be stored in the export package.
145     * @param uri the URI for the resource
146     * @param sourcePath the base path the resource was exported from
147     * @param destinationPath the base path the resource is importing to
148     * @param baseDir the base directory in the export package
149     * @return a unique location for metadata resources contained by the resource at the
150     *         given URI would be stored
151     */
152    public static File directoryForContainer(final URI uri, final String sourcePath,
153            final String destinationPath, final File baseDir) {
154        return fileForURI(uri, sourcePath, destinationPath, baseDir, "");
155    }
156
157    /**
158     * Checks the response code and throws a RuntimeException with a helpful
159     * message (when possible) for non 2xx codes.
160     * @param response the response from a REST call to Fedora
161     * @param uri the URI against which the request was made
162     * @param user the user name for authorization
163     */
164    public static void checkValidResponse(final FcrepoResponse response, final URI uri, final String user) {
165        switch (response.getStatusCode()) {
166        case 401:
167            throw new AuthenticationRequiredRuntimeException();
168        case 403:
169            throw new AuthorizationDeniedRuntimeException(user, uri);
170        case 404:
171            throw new ResourceNotFoundRuntimeException(uri);
172        default:
173            if (response.getStatusCode() < 200 || response.getStatusCode() > 307) {
174                throw new RuntimeException("Export operation failed: unexpected status "
175                        + response.getStatusCode() + " for " + uri);
176            }
177        }
178    }
179
180    /**
181     * Utility method to determine whether the current uri is repository root or not. The repository root is the
182     * container with type fcrepo:RepositoryRoot
183     * @param uri the URI for the resource
184     * @param client the FcrepoClient to query the repository
185     * @param config the Config for import/export
186     * @throws IOException if there is a error with the network connection
187     * @throws FcrepoOperationFailedException if there is a error with Fedora
188     * @return True if the URI is the root of the repository
189     */
190    public static boolean isRepositoryRoot(final URI uri, final FcrepoClient client, final Config config)
191            throws IOException, FcrepoOperationFailedException {
192        final String userName = config.getUsername();
193        final String rdfLanguage = config.getRdfLanguage();
194        try (FcrepoResponse response = client.head(uri).disableRedirects().perform()) {
195            checkValidResponse(response, uri, userName);
196            // The repository root will not be a binary
197            if (response.getLinkHeaders("type").contains(URI.create(NON_RDF_SOURCE.getURI()))) {
198                return false;
199            }
200            try (FcrepoResponse resp = client.get(uri).accept(rdfLanguage).disableRedirects()
201                    .perform()) {
202                final Model model = createDefaultModel().read(resp.getBody(), null, rdfLanguage);
203                if (model.contains(null, RDF_TYPE, REPOSITORY_ROOT)) {
204                    return true;
205                }
206            }
207        }
208        return false;
209    }

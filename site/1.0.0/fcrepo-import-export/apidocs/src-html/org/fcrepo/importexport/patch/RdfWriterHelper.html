018package org.fcrepo.importexport.patch;
019
020import org.apache.jena.atlas.io.AWriter;
021import org.apache.jena.atlas.io.IO;
022import org.apache.jena.atlas.lib.CharSpace;
023import org.apache.jena.riot.RDFFormat;
024import org.apache.jena.riot.RDFWriterRegistry;
025import org.apache.jena.riot.WriterGraphRIOTFactory;
026import org.apache.jena.riot.system.StreamRDFWriter;
027import org.apache.jena.riot.system.StreamRDFWriterFactory;
028import org.fcrepo.jena.FedoraRDFXMLPlainWriter;
029import org.fcrepo.jena.FedoraTurtleWriter;
030import org.fcrepo.jena.FedoraTurtleWriterBlocks;
031import org.fcrepo.jena.FedoraTurtleWriterFlat;
032import org.fcrepo.jena.FedoraWriterStreamRDFBlocks;
033import org.fcrepo.jena.FedoraWriterStreamRDFFlat;
034import org.fcrepo.jena.FedoraWriterStreamRDFPlain;
035import org.slf4j.Logger;
036
037import java.util.Objects;
038
039import static org.slf4j.LoggerFactory.getLogger;
040
041/**
042 * @author awoods
043 * @since 2017/01/03
044 */
045public class RdfWriterHelper {
046
047    private static final Logger LOGGER = getLogger(RdfWriterHelper.class);
048
049    /**
050     * Constructor
051     */
052    public RdfWriterHelper() {
053        // prevent construction
054    }
055
056    private static StreamRDFWriterFactory streamWriterFactoryBlocks =
057            (output, format) -> new FedoraWriterStreamRDFBlocks(output);
058
059    private static StreamRDFWriterFactory streamWriterFactoryFlat =
060            (output, format) -> new FedoraWriterStreamRDFFlat(output);
061
062    private static StreamRDFWriterFactory streamWriterFactoryTriplesQuads =
063            (output, format) -> {
064                final AWriter w = IO.wrapUTF8(output);
065                return new FedoraWriterStreamRDFPlain(w, CharSpace.UTF8);     // N-Quads and N-Triples.
066            };
067
068    private static StreamRDFWriterFactory streamWriterFactoryTriplesQuadsAscii =
069            (output, format) -> {
070                final AWriter w = IO.wrapUTF8(output);
071                return new FedoraWriterStreamRDFPlain(w, CharSpace.ASCII);     // N-Quads and N-Triples.
072            };
073
074    private static WriterGraphRIOTFactory wgfactory = (serialization) -> {
075        if ( Objects.equals(RDFFormat.TURTLE_PRETTY, serialization) ) {
076            return new FedoraTurtleWriter();
077        }
078        if ( Objects.equals(RDFFormat.TURTLE_BLOCKS, serialization) ) {
079            return new FedoraTurtleWriterBlocks();
080        }
081        if ( Objects.equals(RDFFormat.TURTLE_FLAT, serialization) ) {
082            return new FedoraTurtleWriterFlat();
083        }
084        if ( Objects.equals(RDFFormat.RDFXML_PLAIN, serialization) ) {
085            return new FedoraRDFXMLPlainWriter();
086        }
087
088        return null ;
089    };
090
091
092    static {
093        LOGGER.info("Loading JENA 3.1.1 Patched RDF Output Writers");
094        StreamRDFWriter.register(RDFFormat.TURTLE_BLOCKS, streamWriterFactoryBlocks);
095        StreamRDFWriter.register(RDFFormat.TURTLE_FLAT, streamWriterFactoryFlat);
096
097        StreamRDFWriter.register(RDFFormat.NTRIPLES, streamWriterFactoryTriplesQuads);
098        StreamRDFWriter.register(RDFFormat.NTRIPLES_UTF8, streamWriterFactoryTriplesQuads);
099        StreamRDFWriter.register(RDFFormat.NTRIPLES_ASCII, streamWriterFactoryTriplesQuadsAscii);
100
101        RDFWriterRegistry.register(RDFFormat.TURTLE_PRETTY,  wgfactory) ;
102        RDFWriterRegistry.register(RDFFormat.TURTLE_BLOCKS,  wgfactory) ;
103        RDFWriterRegistry.register(RDFFormat.TURTLE_FLAT,    wgfactory) ;
104
105        RDFWriterRegistry.register(RDFFormat.RDFXML_PLAIN,   wgfactory) ;
106    }
107}
